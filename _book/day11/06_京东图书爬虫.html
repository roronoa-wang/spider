6 京东图书爬虫

需求：抓取京东图书的信息

目标：抓取京东图书信息包含: 图书所属大分类、图书所属小的分类、分类的url地址, 图书的名字、封面图片地址、图书url地址、作者、出版社、出版时间、价格

url：https://book.jd.com/booksort.html

思路分析：

  1. 确定数据抓取的入口

    • 数据都在dl下的dt和dd中，其中dt是大分类的标签，dd是小分类的标签

    • 对应的思路可以使选择所有的大分类后，选择他下一个兄弟节点即可

    • xpath中下一个兄弟节点的语法是following-sibling::*[1]

  2. 确定列表页的url地址和程序终止的条件

    有下一页

    没有下一页

  3. 确定数据的位置

    数据大部分都在url地址的响应中，但是价格不在

    通过search all file的方法，找到价格的地址，

    对url地址进行删除无用的参数和url解码之后，获取到价格的url为

    https://p.3.cn/prices/mgets?skuIds=J_11635586613

    其中skuId为商品的id，能够在网页中获取

  4. 使用scrapy的思路完成爬虫，在settings中添加上scrapy_redis的配置

3. 完善爬虫
  • spiders.book.py 文件
# -*- coding: utf-8 -*-
import scrapy
from copy import deepcopy
import json

from jd.items import JdItem


class BookSpider(scrapy.Spider):
    name = 'book'
    allowed_domains = ['jd.com', 'p.3.cn']

    # 起始URL
    start_urls = ['https://book.jd.com/booksort.html']

    def parse(self, response):
        """解析大分类小分类相关数据"""
        # 获取包含大分类的dt列表
        dts = response.xpath('//*[@id="booksort"]/div[2]/dl/dt')

        # 遍历dts, 获取大分类的名称和URL
        for dt in dts:
            # 创建数据模型对象
            item = JdItem()
            # 提取大分类名称和URL
            item['b_category_name'] = dt.xpath('./a/text()').extract_first()
            item['b_category_url'] = 'https:'+ dt.xpath('./a/@href').extract_first()
            # 获取小分类名称和URL
            # 获取包含小分类信息的a标签列表
            a_s = dt.xpath('./following-sibling::dd[1]/em/a')
            # 遍历a_s, 获取小分类信息
            for a in a_s:
                item['s_category_name'] = a.xpath('./text()').extract_first()
                item['s_category_url'] = 'https:' + a.xpath('./@href').extract_first()
                # print(item)
                # 通过小分类的URL构建列表页请求对象, 交给引擎
                yield scrapy.Request(item['s_category_url'], callback=self.parse_book_list, meta={'item': deepcopy(item)})


    def parse_book_list(self, response):
        """解析列表页数据"""
        # 取出上一个解析函数传递过来的数据
        item = response.meta['item']
        # 获取包含图书信息的div列表, 这种写法, 无论是单本或套装的div都可以获取到
        divs = response.xpath('//div[contains(@class, "j-sku-item")]')
        # 遍历divs, 获取图书信息
        for div in divs:
            # 获取图片的URL
            item['book_img_url'] = div.xpath('./div[1]/a/img/@src').extract_first()
            if item['book_img_url'] is None:
                item['book_img_url'] = div.xpath('./div[1]/a/img/@data-lazy-img').extract_first()
            if item['book_img_url']:
                item['book_img_url'] = 'https:' + item['book_img_url']

            # 获取图书名称
            item['book_name'] = div.xpath('./div[3]/a/em/text()').extract_first()
            # 图书URL
            item['book_url'] = div.xpath('./div[3]/a/@href').extract_first()
            if item['book_url']:
                item['book_url'] = 'https:' + item['book_url']

            # 作者
            item['book_author'] = ''.join(div.xpath('./div[4]/span[1]/span/a/text()').extract())
            # 出版社
            item['book_publisher'] = div.xpath('./div[4]/span[2]/a/text()').extract_first()
            # 出版时间
            item['book_published_date'] = div.xpath('./div[4]/span[3]/text()').extract_first()
            # 由于这个方法不是引擎调用的, 而是我们自己内部调用的, 所以这里只能使用 return不能是用yield

            # 构造价格请求
            # 价格URL
            price_url = 'https://p.3.cn/prices/mgets?skuIds=J_{}'
            # 获取书的data-sku, 也就是skuid
            skuid = div.xpath('./@data-sku').extract_first()
            # 生成完整的价格数据
            price_url = price_url.format(skuid)

            # 构建价格请求对象
            yield scrapy.Request(price_url, callback=self.parse_price, meta={'item': deepcopy(item)})

        # 处理列表页的分页
        # 获取下一页的URL
        next_url = response.xpath('//a[@class="pn-next"]/@href').extract_first()
        if next_url:
            next_url = response.urljoin(next_url)
            # print(next_url)
            yield scrapy.Request(next_url, callback=self.parse_book_list, meta={'item': deepcopy(item)})

    def parse_price(self, response):
         # 取出传递过的数据
         item = response.meta['item']
         # 提取价格
         item['book_price'] = json.loads(response.text)[0]['op']
         # 该方法是引擎调用的, 所以通过yield来返回数据
         yield item
